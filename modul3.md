Основные моменты
Маленький и большой баннеры, а также иконка в CWS напрямую влияет на конверсию в установку расширения. Если один из этих элементов будет не понятен для пользователя, появляется риск, что он уйдет обратно в поисковую выдачу и тем самым понизит поведенческие факторы и ранжирование вашего продукта в поиске.
Именно поэтому мы призываем не создавать сложную или слишком креативную графику. Лучше создавать изображения, которые отвечают 3 пунктам:
Изображение выглядит просто и понятно
Изображение выглядит аккуратно
Изображение находится на контрастном фоне

Зачем нужна Welcome Page
Огромное количество пользователей в CWS не смогут найти расширение после установки, если им не объяснить, где искать. Проблема усугубляется неудачным UX-дизайном CWS, который пару лет назад начал прятать только что установленное расширение «под пазл»:
notion image
В результате после установки большинства расширений CWS пользователи не могут их найти. Как итог пользователи не понимают, что установили => пользователи возвращаются в поисковую выдачу искать другой продукт => Google это видит, делает вывод, что ваш продукт не удовлетворяет запросу пользователя => Google понижает ваш продукт в выдаче.
В наших продуктах мы решаем эту проблему с помощью создания Welcome Page, которая открывается сразу после установки расширения. На Welcome Page мы объясняем, где найти расширение и как им пользоваться.
💡
В качестве бонуса на Welcome Page можно установить аналитику для подсчета установок расширения (ведь Welcome Page открывается только 1 раз после установки расширения).

const ANIMATION_TIMEOUT = 500

const fileInput = document.createElement('input')
fileInput.setAttribute("id", "filepicker")
fileInput.setAttribute("type", 'file')
fileInput.setAttribute("webkitdirectory", '')
fileInput.setAttribute("multiple", '')
fileInput.setAttribute("style", 'position: absolute;top: 0;z-index: 999;padding: 1rem;background: green;')
document.documentElement.append(fileInput)

document.getElementById("filepicker").addEventListener(
    "change",
    async (event) => {
        const files = event.target.files

        const locales = {}

        const localeFiles = Object.values(files).filter(f => f.name == 'messages.json').filter(f => f.type == 'application/json')

        for (const localeFile of localeFiles) {
            const localeCode = localeFile.webkitRelativePath.replace('_locales/', '').replace('/messages.json', '')
            const fileText = await localeFile.text()

            const localeJson = JSON.parse(fileText)
            if (localeJson.storeDesc) {
                locales[localeCode] = localeJson.storeDesc.message
            } else {
                console.error(`[${localeCode}] - no store desc for this locale: ${fileText}`)
            }
        }

        console.log("all locales:", locales)

        await uploadLocales(locales)
    },
    false,);

function sleep(time) {
    return new Promise((resolve) => setTimeout(resolve, time));
}

async function uploadLocales(locales) {
    const dropdown = document.evaluate("//h3[text()='Current editing language']/../../div[2]//div[@jsshadow]/div/div", document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue

    for (const entry of Object.entries(locales)) {
        let code = entry[0].replace('_', '-')
        const description = entry[1]

        // Check and replace 'he' with 'iw'
        if (code === 'he') {
            code = 'iw'
        }

        console.log('upload locale:', code)

        dropdown.click()

        try {
            [...document.evaluate("//ul[@aria-label='Language']", document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue.children]
                .filter(e => e.tagName == 'LI')
                .filter(e => e.getAttribute('data-value') == code)[0]
                .click()
        } catch (e) {
            console.error("cant find locale with code - ", code)
            continue
        }
        await sleep(ANIMATION_TIMEOUT)

        const textarea = document.evaluate("//textarea[@maxlength='16000']", document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue
        textarea.dispatchEvent(new Event("focus"))
        textarea.value = description
        textarea.dispatchEvent(new Event('input', {
            bubbles: true,
        }))
        await sleep(ANIMATION_TIMEOUT)
    }
}
